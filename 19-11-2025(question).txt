        self.Titile=Title

        self.Genre=Genre

    def diplay(self):
        print("Title:",self.Titile)
        print("Author:",self.Author)
        print("Publis_year:",self.Publish_year)
        print("Available:",self.Available)
        print("Genre:",self.Genre)
